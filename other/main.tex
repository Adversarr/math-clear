\end{definition}


\begin{theorem}{阿基米德性质}{Archimedean-property}
	设 $a,\varepsilon \in \mathbb R$ 为正数，则存在正整数 $M > 0$ 使得 $a < M\varepsilon$。
\end{theorem}
